// 1. ВСТАВЬТЕ СЮДА ID ВАШЕЙ ТАБЛИЦЫ
// ═══════════════════════════════════════════
const SPREADSHEET_ID = 'ВСТАВЬТЕ_ID_ТАБЛИЦЫ';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    console.log('📥 Получен запрос:', JSON.stringify(data));

    // Читаем настройки из таблицы
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

    // Токен бота из Settings
    const settingsSheet = ss.getSheetByName('Settings');
    const settingsData = settingsSheet.getDataRange().getValues();
    let token = '';
    for (let i = 0; i < settingsData.length; i++) {
      if (settingsData[i][0] === 'bot_token') {
        token = settingsData[i][1].toString().trim();
        break;
      }
    }

    if (!token) {
      throw new Error('Токен не найден в таблице Settings');
    }
    console.log('🔑 Токен загружен');

    // Читаем пользователей из Users
    const usersSheet = ss.getSheetByName('Users');
    const usersData = usersSheet.getDataRange().getValues();
    const headers = usersData[0];

    // Индексы колонок
    const chatIdIdx = headers.indexOf('Chat ID');
    const nameIdx = headers.indexOf('Name');
    const feedbackIdx = headers.indexOf('Feedback');
    const bookingIdx = headers.indexOf('Booking');
    const franchiseIdx = headers.indexOf('Franchise');
    const activeIdx = headers.indexOf('Active');

    if (chatIdIdx === -1 || activeIdx === -1) {
      throw new Error('Неверная структура таблицы. Нужны колонки: Chat ID, Name, Feedback, Booking, Franchise, Active');
    }

    // Формируем текст сообщения
    let text = '';
    switch(data.type) {
      case 'feedback':
        text = '✉️ *Новый отзыв!*\n👤 Имя: ' + data.name + '\n📞 Телефон: ' + data.phone + '\n⭐ Рейтинг: ' + (data.rating || 'не указан') + '\n💬 ' + data.message;
        break;
      case 'booking':
        text = '🍽 *Бронирование банкета*\n👤 Имя: ' + data.name + '\n📞 Телефон: ' + data.phone + '\n📅 Дата: ' + data.date + '\n👥 Гостей: ' + data.guests + '\n📝 Пожелания: ' + (data.notes || 'нет');
        break;
      case 'franchise':
        text = '📊 *Заявка на франшизу*\n👤 Имя: ' + data.name + '\n📞 Телефон: ' + data.phone + '\n📧 Email: ' + data.email + '\n🏙 Город: ' + data.city;
        break;
      default:
        text = '❓ Неизвестный тип: ' + data.type;
    }

    // Отправляем всем подходящим пользователям
    let sentCount = 0;
    for (let i = 1; i < usersData.length; i++) {
      const row = usersData[i];
      const isActive = String(row[activeIdx]).toUpperCase() === 'TRUE' || row[activeIdx] === true;

      if (!isActive) continue;

      const chatId = String(row[chatIdIdx]).trim();
      if (!chatId) continue;

      // Проверяем права на этот тип уведомления
      let hasPermission = false;
      if (data.type === 'feedback' && feedbackIdx !== -1) {
        hasPermission = String(row[feedbackIdx]).toUpperCase() === 'TRUE' || row[feedbackIdx] === true;
      } else if (data.type === 'booking' && bookingIdx !== -1) {
        hasPermission = String(row[bookingIdx]).toUpperCase() === 'TRUE' || row[bookingIdx] === true;
      } else if (data.type === 'franchise' && franchiseIdx !== -1) {
        hasPermission = String(row[franchiseIdx]).toUpperCase() === 'TRUE' || row[franchiseIdx] === true;
      }

      if (!hasPermission) continue;

      try {
        const result = UrlFetchApp.fetch('https://api.telegram.org/bot' + token + '/sendMessage', {
          method: 'post',
          payload: {
            chat_id: chatId,
            text: text,
            parse_mode: 'Markdown'
          }
        });
        console.log('✅ Отправлено пользователю ' + chatId + ': ' + result.getContentText());
        sentCount++;
      } catch (sendError) {
        console.error('❌ Ошибка отправки пользователю ' + chatId + ': ' + sendError.message);
      }
    }

    console.log('📊 Отправлено ' + sentCount + ' пользователям');

    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', sent: sentCount }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    console.error('❌ Ошибка:', error.message);
    console.error('Стек:', error.stack);

    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
